\AtBeginSection[]{ \begin{frame}\frametitle{\thesection. \insertsectionhead} % セクション番号とセクション名を表示
\tableofcontents[currentsection, hideothersubsections] % 該当セクションだけ強調表示, 他のsubsectionは表示しない
\end{frame} }

\AtBeginSubsection[]{ \begin{frame}\frametitle{\thesection.\thesubsection \insertsubsectionhead} % セクション番号.サブセクション番号 サブセクション名を表示
\tableofcontents[currentsection, currentsubsection] % 該当セクションとサブセクションを強調
\end{frame}}
